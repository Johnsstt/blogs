## 顺序栈
```c++
const int maxSize=4;

typedef struct{
	int data[maxSize];
	int top; // 栈顶指针 
}SqStack;

// 初始化栈
void initStack(SqStack st){
 	st.top=-1;
 }
// 判断栈空
int isEmpty(SqStack st){
	return st.top==-1?1:0;
} 
// 进栈 
int push(SqStack &st,int x){
	if(st.top==maxSize-1){
		cout<<"栈满，不能进栈"<<endl;
		return 0;
	}else{
		st.data[++st.top]=x;
		return 1;
	}
}
// 出栈
int pop(SqStack &st){
	if(st.top==-1){
		cout<<"栈空，不能进行出栈操作"<<endl;
	}else{
		return st.data[(st.top)--];
	}
} 

int main(){
	SqStack st;
	initStack(st);
	push(st,8);
	push(st,5);
	push(st,3);
	pop(st);
	
}
```
## 链式栈
操作和单链如出一辙
```c++
/* 链式栈*/
// 初始化 
void initStack(Node *&st){
	st=(Node*)malloc(sizeof(Node));
	st->next=NULL;
}
// 判断栈空 
int isEmpty(Node *st){
	return st->next==NULL?1:0;
}
// 入栈 
void push(Node *st,int x){
	Node *p;
	p=(Node*)malloc(sizeof(Node));
	p->data=x;
	p->next=st->next;
	st->next=p;
}
// 出栈
int pop(Node *st){
	if(st->next==NULL) return 0;
	Node *p;
	p=st->next;
	st->next=p->next;
	free(p);
	return 1;
}
```
## 栈的应用
### 表达式中的括号是否合法——左右括号匹配
```cpp
const int maxSize=99;

int match(char exp[],int n){
	char stack[maxSize];int top=-1;
	for(int i=0;i<n;i++){
		if(exp[i]=='('){
			stack[++top]='(';
		}else if(exp[i]==')'){
			if(top==-1){
				return 0;
			}
			top--;
		}
	}
	if(top==-1){
		return 1;
	}
	return 0;
}


int main(){
	char exp[8]={"(1+2))"};
	int n=6;
	int res=match(exp,n);
	cout<<res; // 0
}
```
### 中缀、前缀、后缀表达式求值
1. 先将前缀或后缀表达式的数字压入堆栈，前缀从右向左扫描，后缀从左向右扫描。
2. 遇到操作符，弹出栈顶两个元素，进行相应计算，之后将结果数值压入栈中。
3. 扫描到结束，即求助表达式的结果。
#### 后缀
```cpp
const int maxSize=99;
int operate(char op,int a,int b){
	switch(op){
		case '+': return a+b;
		case '-': return a-b;
		case '*': return a*b;
		case '/': {
			if(b==0){
				cout<<"error"<<endl;
				return 0;
			}
			return a/b;
		}
		default: {
			cout<<"error operator"<<endl;
			return 0;
		}
	}
}

int evalute(char exp[]){
	int stack[maxSize];int top=-1;
	char Op;
	for(int i=0;exp[i]!='\0';i++){
		if(exp[i]>='0'&&exp[i]<='9'){
			stack[++top]=exp[i]-'0';
		}else{
			Op=exp[i];
			int num=operate(Op,stack[top--],stack[top--]);
			stack[++top]=num;
		}
	}
	return stack[top];
}


int main(){
	char exp[8]={"12+3*"};
	int n=6;
	int res=evalute(exp);
	cout<<res; // 0
}
```
### 数制转换（十进制转为其他进制）
```cpp
void trans(int num,int decimal){
	int stack[maxSize];int top=-1;
	int n=num;
	while(n){
		stack[++top]=n%decimal;
		n=n/decimal;
	}
	while(top!=-1){
		cout<<stack[top--];
	}
}


int main(){
	trans(13,2);
}
```
