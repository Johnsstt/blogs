## 顺序栈
```c++
const int maxSize=4;

typedef struct{
	int data[maxSize];
	int top; // 栈顶指针 
}SqStack;

// 初始化栈
void initStack(SqStack st){
 	st.top=-1;
 }
// 判断栈空
int isEmpty(SqStack st){
	return st.top==-1?1:0;
} 
// 进栈 
int push(SqStack &st,int x){
	if(st.top==maxSize-1){
		cout<<"栈满，不能进栈"<<endl;
		return 0;
	}else{
		st.data[++st.top]=x;
		return 1;
	}
}
// 出栈
int pop(SqStack &st){
	if(st.top==-1){
		cout<<"栈空，不能进行出栈操作"<<endl;
	}else{
		return st.data[(st.top)--];
	}
} 

int main(){
	SqStack st;
	initStack(st);
	push(st,8);
	push(st,5);
	push(st,3);
	pop(st);
	
}
```
## 链式栈
操作和单链如出一辙
```c++
/* 链式栈*/
// 初始化 
void initStack(Node *&st){
	st=(Node*)malloc(sizeof(Node));
	st->next=NULL;
}
// 判断栈空 
int isEmpty(Node *st){
	return st->next==NULL?1:0;
}
// 入栈 
void push(Node *st,int x){
	Node *p;
	p=(Node*)malloc(sizeof(Node));
	p->data=x;
	p->next=st->next;
	st->next=p;
}
// 出栈
int pop(Node *st){
	if(st->next==NULL) return 0;
	Node *p;
	p=st->next;
	st->next=p->next;
	free(p);
	return 1;
}
```
